---
title: Button Group
description: Displays a set of buttons - where no more than one of the buttons can be checked at a time.
---

import { Preview } from "../../components/Preview";
import { ButtonGroup, ButtonGroupItem } from "natmfat";
import { tokens } from "natmfat/lib/tokens";

## Preview

<Preview>
  <ButtonGroup
    style={{ width: tokens.space256 }}
    orientation="horizontal"
    defaultValue="unread"
    color="primary"
  >
    <ButtonGroupItem value="unread">Unread</ButtonGroupItem>
    <ButtonGroupItem value="all">All</ButtonGroupItem>
  </ButtonGroup>

## Usage

```jsx
import { ButtonGroup, ButtonGroupItem } from "natmfat/components/ButtonGroup";
import { tokens } from "natmfat/lib/tokens";
```

```jsx
<ButtonGroup
  style={{ width: tokens.space256 }}
  orientation="horizontal"
  defaultValue="unread"
  color="primary"
>
  <ButtonGroupItem value="unread">Unread</ButtonGroupItem>
  <ButtonGroupItem value="all">All</ButtonGroupItem>
</ButtonGroup>
```
